Add the Auditor Sentiment NLP ModuleSince downloading 10 years of PDFs live takes >90 seconds, you must build a "Sentiment Engine" that scans whatever textual notes you can get (or builds the framework to show the judges how it works).Add this class above your ReportGenerator:Pythonclass AuditorSentimentAnalyzer:
    """Analyzes the language of auditor qualifications over time."""
    
    RED_FLAG_WORDS = ['subject to', 'uncertainty', 'material misstatement', 
                      'emphasis of matter', 'except for', 'adverse', 'going concern',
                      'discrepancy', 'fraud', 'restate']
                      
    STABLE_WORDS = ['true and fair view', 'unmodified', 'compliance', 
                    'materially correct', 'fair presentation']

    def analyze_sentiment_trend(self, notes_data_dict):
        """
        Calculates a hedged language score.
        (Note: For the hackathon demo, if live PDF parsing is too slow, 
        you can use this logic to score pre-cached text snippets).
        """
        sentiment_trend = {}
        for year, text in notes_data_dict.items():
            text_lower = str(text).lower()
            red_flags = sum(text_lower.count(word) for word in self.RED_FLAG_WORDS)
            stable = sum(text_lower.count(word) for word in self.STABLE_WORDS)
            
            # Sentiment score: Higher is more hedged/risky
            score = (red_flags * 2) - stable
            sentiment_trend[year] = {
                'score': score,
                'hedged_keywords_found': [w for w in self.RED_FLAG_WORDS if w in text_lower]
            }
        return sentiment_trend
